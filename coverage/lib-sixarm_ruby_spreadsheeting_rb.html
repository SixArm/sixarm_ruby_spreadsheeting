31  # Parse various input words (e.g. "Yes", "No", "On", "Off") to true or false.
 
32  #
 
33  # There's a typical import use case where a column is
 
34  # a boolean, in a variety of formats like "X" for on,
 
35  # "Y" for yes, "T" for true, etc.
 
36  #
 
37  # @param [Boolean, String, #to_s, nil] input - typically "Yes", "No", "On", "Off", etc.
 
38  # @return [Boolean] the parsed output
 
39  #
 
40  # We use these rules:
 
41  # - x y yes t true on +  => true
 
42  # - blank n no f false off -  => false
 
43  # - anything else => nil
 
44  #
 
45  # @example
 
46  #   import_flag("Yes") => true
 
47  #   import_flag("No") => false
 
48  #   import_flag("X") => true
 
49  #   import_flag("") => false
 
50  #   import_flag("Hello") => nil
 
51  #   import_flag(true) => true
 
52  #   import_flag(false) => true
 
53  #   import_flag(nil) => nil
 
54  #
 
55  # @see #import_flag_or_note
 
56 
57  def import_flag(input)
 
58   return input if (input==true or input==false or input==nil)
 
59   case input.to_s.strip.downcase
 
60   when 'x','y','yes','t','true','on','+'
 
61    return true
 
62   when '','n','no','f','false','off','-'
 
63    return false
 
64   else 
65    return nil
 
66   end
 
67  end

70  # Parse an imput that may be a flag (e.g. true/false) or a note (e.g. freeform text) into an array.
 
71  #
 
72  # There's a typical import use case where a column
 
73  # can contain two different data types:
 
74  # - a boolean, in a variety of formats (see _import_flag_)
 
75  # - a note, in plain text
 
76  #
 
77  # We need to separate these two.
 
78  #
 
79  # @param [Boolean, String, #to_s, nil] input - typically "Yes", "No", "On", "Off", etc.
 
80  # @return [Array<Boolean, String>] the parsed output [flag, note]
 
81  #
 
82  # We use these rules:
 
83  # - x y yes t true on +  => [true, '']
 
84  # - blank n no f false off -  => [false, '']
 
85  # - any other string => [nil, note]
 
86  # - nil => [nil, nil]
 
87  #
 
88  # @example
 
89  #   import_flag_or_note("Yes") => [true,""]
 
90  #   import_flag_or_note("No") => [false,""]
 
91  #   import_flag_or_note("X") => [true,""]
 
92  #   import_flag_or_note("") => [false,""]
 
93  #   import_flag_or_note("Hello") => [nil,"Hello"]
 
94  #   import_flag_or_note(true) => [true,""]
 
95  #   import_flag_or_note(false) => [false,""]
 
96  #   import_flag_or_note(nil) => [nil,nil]
 
97  #
 
98  # @see #export_flag_or_note
 
99  # @see #import_flag
 
100 
101  def import_flag_or_note(s)
 
102    return [nil, nil] if (s==nil)
 
103    flag = import_flag(s)
 
104    return [flag, flag==nil ? s : '']
 
105  end

108  # Parse an imput that may be a date (e.g. "2007/12/31") or a note (e.g. freeform text) into an array.
 
109  #
 
110  # There's a common import use case where a column can be of two types:
 
111  # - a date, in a variety of format like "Jan 1, 2008", "1/1/2008", etc.
 
112  # - a note, in plain text
 
113  #
 
114  # @param [Boolean, String, #to_s, nil] input - typically "Yes", "No", "On", "Off", etc.
 
115  # @return [Array<Date, String>] the parsed output [date, note]
 
116  #
 
117  # Examples:
 
118  #   import_date_or_note('1/1/2007') => [Date(2007,01,01),'']
 
119  #   import_date_or_note('hello') => [nil,'hello']
 
120  #
 
121  # @see #export_date_or_note
 
122  # @see #import_date
 
123  
124  def import_date_or_note(s)
 
125   begin
 
126    d=import_date(s)
 
127   rescue
 
128    d=nil
 
129   end
 
130   return d ? [d,''] : [nil,s]
 
131  end
